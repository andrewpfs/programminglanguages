package numbergenerator;

import java.lang.Math;
import java.util.Scanner;

public class App {
    public static void numberGuesser(int number) {
      numberGuesser(number , 0);
    }
    public static String numberGuesser(int number, int attempts){
      Scanner reader = new Scanner(System.in);
      String resultString = new String();
      resultString += ("You have guessed " + attempts + " times");
      System.out.print("What is your guess: ");
      int guess = reader.nextInt();
      String result = guessChecker(guess, number);
      if(result.equals("Lower")){
        resultString += ("Your guess was lower than the number");
        numberGuesser(number, attempts+1);
      }
      else if(result.equals("Higher")){
        resultString += ("Your guess was higher than the number");
        numberGuesser(number,attempts+1);
      }
      else{
        resultString +=("You have guessed correctly! The number was " + number + "!");
        resultString += ("It took you " + attempts+1 + " guesses");
      }
      return resultString;
    }
    public static String guessChecker(int guess, int number){
      if(guess == number){
        return "Equal";
      }
      else if (guess < number){
        return "Lower";
      }
      else {
        return "Higher";
      }
    }
    public static void main(String[]args){
      int number = (int)(Math.random() * 100);
      System.out.println(number +"\n");
      numberGuesser(number);
    }
}
